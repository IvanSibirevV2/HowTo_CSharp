CREATE TABLE Комната (
    id       INTEGER PRIMARY KEY ASC AUTOINCREMENT,
    Название STRING
);
CREATE TABLE Шакаф (
    id       INTEGER PRIMARY KEY ASC AUTOINCREMENT,
    Название STRING,
    id_Комната    INTEGER REFERENCES Комната (id) 
);
CREATE TABLE Полка (
    id       INTEGER PRIMARY KEY ASC AUTOINCREMENT,
    Название STRING,
    id_Шакаф    INTEGER REFERENCES Шакаф (id) 
);
CREATE TABLE ТипПроцессора (
    id       INTEGER PRIMARY KEY ASC AUTOINCREMENT,
    Название STRING,
	Характеристики STRING
);
CREATE TABLE ЭкземплярХранимогоПроцессора (
    id       INTEGER PRIMARY KEY ASC AUTOINCREMENT,
    id_ТипПроцессора    INTEGER REFERENCES ТипПроцессора (id),
	id_Полка    INTEGER REFERENCES Полка (id) 
);
